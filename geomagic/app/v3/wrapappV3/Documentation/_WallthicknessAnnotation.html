Examples

Get all the measurements from the active model

Get all the measurements from the active model

import geomagic.app.v3
from geomagic.app.v3.imports import *

# get the active model and find all measurements

import geomagic.app.v3
from geomagic.app.v3.imports import *
from geomagic.app.v3.wrapappV3 import *

model = geoapp.getActiveModel()
gm = GetMeasurementsFromModel()
gm.model = model
gm.run()

for m in gm.measurements:
    print ("name: " + m.name)
    print ("value: " + str(m.value))
    print ("tolerance: " + str(m.tolerance))
    if m.result == Measurement.PASS:
        print ("status: PASS")
    elif m.result == Measurement.FAIL:
        print ("status: FAIL")
    else:
        print ("status: INVALID")

    if isinstance(m, Dimension):
        print ("Dimension:")
        print ("    viewName: " + m.viewName)
        print ("    dimType: " + m.dimensionType)
        print ("    measured: " + str(m.measured))
        print ("    nominal: " + str(m.nominal))
        print ("    upperTol: " + str(m.upperTolerance))
        print ("    lowerTol: " + str(m.lowerTolerance))
    elif isinstance(m, GDTCallout):
        print ("Callout:")
        print ("    viewName: " + m.viewName)
        print ("    dimType: " + m.dimensionType)
        print ("    bonusTolerance: " + str(m.bonusTolerance))
        print ("    minimumValue: " + str(m.minimumValue))
        print ("    maximumValue: " + str(m.maximumValue))
    elif isinstance(m, DeviationAnnotation):
        print ("Annotation:")
        print ("    viewName: " + m.viewName)
        print ("    upperTol: " + str(m.upperTolerance))
        print ("    lowerTol: " + str(m.lowerTolerance))
        print ("    deviationRadius: " + str(m.deviationRadius))
        print ("    normal: " + str(m.normal))
        print ("    measured: " + str(m.measured))
        print ("    nominal: " + str(m.nominal))
    elif isinstance(m, FeatureAnnotation):
        print ("Feature:")
        print ("    viewName: " + m.viewName)
        print ("    featureType: " + m.featureType)
        print ("    form: " + str(m.form))
    elif isinstance(m, WallthicknessAnnotation):
        print ("Wallthickness:")
        print ("    viewName: " + m.viewName)
        print ("    upperTol: " + str(m.upperTolerance))
        print ("    lowerTol: " + str(m.lowerTolerance))
        print ("    measured: " + str(m.measured))
        print ("    nominal: " + str(m.nominal))
