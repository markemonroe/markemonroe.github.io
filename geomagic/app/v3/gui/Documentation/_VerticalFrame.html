VerticalFrame¶

Presents a frame container for other widgets. Frames may be visible or invisible. They may also be collapsible and have a label. Vertical frames present the contained widgets from top to bottom in order that they were added to the frame. If the frame is expanded or collapsed, notifications can be received by subscribing to the stateChanged(sender, value) event.

addWidget(widget)¶

Add a widget to the content of the frame. A widget is an instance of any of the classes in the geoapp.gui.widgets namespace.

Examples

Create a basic vertical frame and add it to a dialog.

import geomagic.app.v3
from geomagic.app.v3.imports import *

class BasicVerticalFrameDialogHandler(guiDlg.DialogHandler):

   # Constructor. You must call the base class constructor.
   def __init__(self, name, modal):        
      super(BasicVerticalFrameDialogHandler, self).__init__(name, modal)
   
   # All Widgets must be created in formDialog.
   def formDialog(self):
      
      # Get the top Frame to add widgets to.
      topFrame = self.dialog().topFrame()
      
      # Create a Vertical Frame and add it to the top frame.
      self.verticalFrame = guiWidgets.VerticalFrame(u'Vertical Frame', u'Vertical Frame', True)
      topFrame.addWidget(self.verticalFrame)
      
      # Create a FloatWidget and add it to the vertical frame
      self.floatWidget = guiWidgets.FloatWidget(u'Float Widget', u'Float Widget')
      self.verticalFrame.addWidget(self.floatWidget)
      
      return True
dialog = BasicVerticalFrameDialogHandler(u'Basic Vertical Frame', False)
dialog.start()

Create an advanced vertical frame with nested frames and add it to a dialog.

import geomagic.app.v3
from geomagic.app.v3.imports import *

class AdvancedVerticalFrameDialogHandler(guiDlg.DialogHandler):
   
   # Constructor. You must call the base class constructor.
   def __init__(self, name, modal):        
      super(AdvancedVerticalFrameDialogHandler, self).__init__(name, modal)
      
   # All Widgets must be created in formDialog
   def formDialog(self):
      # Get the top Frame to add widgets to.
      topFrame = self.dialog().topFrame()
      # Create a Vertical Frame Widget and add it to the top frame.
      self.outerFrame = guiWidgets.VerticalFrame(u'Outer Vertical Frame', u'Outer Vertical Frame', True)
      self.outerFrame.setPadding(gui.Thickness(10))
      self.outerFrame.setMargin(gui.Thickness(5))
      topFrame.addWidget(self.outerFrame)
      
      # Create a Vertical Frame and add it to the outer frame
      self.innerFrame = guiWidgets.VerticalFrame(u'Inner Vertical Frame', u'Inner Vertical Frame', True)
      self.outerFrame.addWidget(self.innerFrame)
      self.label1 = guiWidgets.LabelWidget('Label 1', u'Label 1')
      self.label2 = guiWidgets.LabelWidget('Label 2', u'Label 2')
      self.innerFrame.addWidget(self.label1)
      self.innerFrame.addWidget(self.label2)
      
      # Create another Vertical Frame and add it to the outer frame.  Override the
      # default nested behavior making the frame an expander.
      self.innerFrame2 = guiWidgets.VerticalFrame(u'Inner Vertical Frame 2', u'Inner Vertical Frame 2', True, True)
      self.innerFrame2.setExpanded(False)
      self.outerFrame.addWidget(self.innerFrame2)
      self.label3 = guiWidgets.LabelWidget('Label 3', u'Label 3')
      self.label4 = guiWidgets.LabelWidget('Label 4', u'Label 4')
      self.innerFrame2.addWidget(self.label3)
      self.innerFrame2.addWidget(self.label4)
      
      return True
      
dialog = AdvancedVerticalFrameDialogHandler(u'Advanced Vertical Frame', False)
dialog.start()
