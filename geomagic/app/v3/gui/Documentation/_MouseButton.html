Examples

Create a dialog that monitors mouse button states.

import geomagic.app.v3
from geomagic.app.v3.imports import *
class MouseButtonDialogHandler(guiDlg.DialogHandler):
   # Constructor. You must call the base class constructor.
   def __init__(self, name, modal):
      super(MouseButtonDialogHandler, self).__init__(name, modal)
      self.dlg = self.dialog()
      
   # All Widgets must be created in formDialog.
   def formDialog(self):
      # Create a mouse Handler and add it to the dialog.
      self.mh = guiMouse.MouseHandler()
      self.dlg.setMouseHandler(self.mh)

      # Get the top Frame to add widgets to.
      topFrame = self.dlg.topFrame()
      
      # Create a vertical frame to add widgets to.
      self.vertFrame = guiWidgets.VerticalFrame('VerticalFrameStatus', 'Status', True, False)
      topFrame.addWidget(self.vertFrame)

      # Create 2 TextWidgets to output the results   
      self.mb = guiWidgets.TextWidget('mb', u'Button')
      self.vertFrame.addWidget(self.mb)

      self.tw = guiWidgets.TextWidget('tw', u'State')
      self.vertFrame.addWidget(self.tw)
      
      return True
      
   def onStart(self):
      # Subscribe to the mouse events.
      self.mh.mousePressed += self.onMousePressed
      self.mh.mouseReleased += self.onMouseReleased
      
      # Initialize the Dialog.
      self.mb.setText(u'None')
      self.tw.setText(u'Up')
      
      return True
      
   def onStop(self):
      # Unsubscribe to the mouse events.
      self.mh.mousePressed -= self.onMousePressed
      self.mh.mouseReleased -= self.onMouseReleased
      
   def onMousePressed(self, sender, mouseData):
      # update the dialog      
      btnCode = mouseData.getMode()
      btnText = u'None'
      if btnCode == guiMouse.MouseButton.Left:
         btnText = u'Left'
      elif btnCode == guiMouse.MouseButton.Middle:
         btnText = u'Middle'
      elif btnCode == guiMouse.MouseButton.Right:
         btnText = u'Right'
         
      self.mb.setText(btnText)
      self.tw.setText(u'Down')
   
   def onMouseReleased(self, sender, mouseData):
      # update the dialog
      self.mb.setText(u'None')
      self.tw.setText(u'Up')
      
# Now start the dialog
dialog = MouseButtonDialogHandler('Mouse Button Dialog', False)
dialog.start()
