Table¶

Examples

Create a table.

from geomagic.api.v3 import *

#create table object
table = Table()

# init table with 3 columns and title and label text
table.init(3, "test table label", "test table title")
table.title = "new title"

print("label: " + str(table.title))

#add row
rownum = table.addRow()
print("row id: " + str(rownum))

# addcells
nameOfWidget = "widget1"
colid = 0
table.setCell(rownum, colid, nameOfWidget)

descOfWidget = "metal widget"
colid = 1
table.setCell(rownum, colid, descOfWidget)

numberOfWidgets = 3
colid = 2
table.setCell(rownum, colid, numberOfWidgets)
