Examples

Create Correspondence between Mesh and Points.

from geomagic.api.v3 import *

# Create mesh and points
mesh = openCone()
ptCover = CreatePointCover(mesh)
ptCover.run()
points = ptCover.points

# Create Correspondence
creator = CreateCorrespondence()
creator.testObj = points
creator.refObj = mesh
creator.useNormals = Correspondence.Normals_UseOriented
creator.maxDeviation = 1e-5
creator.run()

correspondence = creator.correspondence
