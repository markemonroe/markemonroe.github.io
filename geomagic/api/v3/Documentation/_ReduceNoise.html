Examples

Smooth a noisy point set using Quadric method.

from geomagic.api.v3 import *
import random

# Create a point set with noise
pts = Points()
def noise(x,y):
   return float(random.randint(-100,100))/1000.0
   
pts = createHeightMapOfPoints(4, noise)

# Create a selection
selection = PointSelection(pts)
selection.selectAll()

# Reduce noise from the selection using Quadratic method
smoother = ReduceNoise()   
smoother.selection = selection
smoother.type = ReduceNoise.Quadratic
smoother.smoothingLevel = 3
smoother.maximumDeviation = 3.14
smoother.run()
