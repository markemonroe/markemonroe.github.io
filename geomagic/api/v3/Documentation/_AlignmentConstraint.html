AlignmentConstraint¶

Defines the result of a feature based alignment algorithm (FeatureAlignment). It can be used as an optional input parameter for best-fit alignment modifiers (BestFitAlign or BestFitAlign2D) or RPS alignment algorithm (AlignUsingReferencePoints).

Examples

A best-fit 3D alignment with a constraint is called.

from geomagic.api.v3 import *

#
# Setting up a rotational axis type of freedom around the z-axis,
# and a translational freedom along z. The movement is
# locked down (constrained) in any other degree of freedom.
#
cst = AlignmentConstraint()
cst.setValidDof(AlignmentConstraint.zRotate, True)
cst.setValidDof(AlignmentConstraint.zTranslate, True)
cst.setValidDof(AlignmentConstraint.xRotate, False)
cst.setValidDof(AlignmentConstraint.xTranslate, False)
cst.setValidDof(AlignmentConstraint.yRotate, False)
cst.setValidDof(AlignmentConstraint.yTranslate, False)
#
# Coordinate system
#
csys = CoordinateSystem()
cx = Vector3D(0,1,0)
cy = Vector3D(-1,0,0)
csys.initialize(Vector3D(0,0,0), cx, cy)
#
# Setting the coordinate system into the constraint:
cst.CSys = csys
#
# Create sample mesh. This will be used as the fixedObject
refMesh = createRefinedTetrahedron()

# Create points from mesh. This will be used as the floatingObject
testPoints = createPointsFromMesh(refMesh)

# Transform the floatingObject a bit so that the two models
# need to be aligned.
rotvec = Vector3D(0, 0, 1)
rotmat = Transform3D()
rotmat.setRotationTransform(rotvec, 0.05)
testPoints.transform(rotmat)
# Align fixedObject and floatingObject using BestFitAlign.
bfAlign = BestFitAlign()
bfAlign.fineAdjustmentsOnly = False
bfAlign.sampleSize = 300
bfAlign.tolerance = 0.0001
bfAlign.fixedObject = refMesh
bfAlign.floatingObject = testPoints
# Constraint:
bfAlign.targetConstraint = cst
#
bfAlign.run()
# distance needed to move test to get it aligned to ref.
dist1 = bfAlign.distance
