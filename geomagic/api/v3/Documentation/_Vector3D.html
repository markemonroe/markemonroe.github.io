Examples

This example shows the creation and use of a Vector3D object

from geomagic.api.v3 import *

# Create Vector3D object
v = Vector3D(1.0, 0.0, 0.0)

# Set using array operator
v[0] = 1.0
v[1] = 2.0
v[2] = 3.0

# Get using array operator
x = v[0]
y = v[1]
z = v[2]

# Get using accessors
z = v.x()
y = v.y()
z = v.z()

# Cross product of two vectors
xAxis = Vector3D(1.0, 0.0, 0.0)
yAxis = Vector3D(0.0, 1.0, 0.0)
zAxis = xAxis * yAxis

# Make vector unit length
zAxis.normalize()

# Dot Product
v1 = Vector3D(2.0, 4.0, 6.0)
v3 = Vector3D(1.0, 1.0, 1.0)
val = v1.dot(v3)

# Create two vectors and add them together.
vec1 = Vector3D(1.0, 1.0, 1.0)
vec2 = Vector3D(1.0, 1.0, 1.0)
vec3 = vec1 + vec2
