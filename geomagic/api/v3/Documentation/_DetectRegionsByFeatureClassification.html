Examples

Run DetectRegionsByFeatureClassification on a cube to divide it into regions

from geomagic.api.v3 import *

# Start with a mesh of a cube with rounded edges
mesh = createBox(2)

# Use region detection to create a regionset with default parameters
detectMod = DetectRegionsByFeatureClassification()
detectMod.mesh = mesh
detectMod.run()
regionSet1 = detectMod.regionSet

# Inspect the properties of the regionSet.  We will have six primary regions (one for each face)
print("The regionset has " + str(len(regionSet1.regions)) + " regions.")
print("The regionset has " + str(len(regionSet1.secondaryRegions)) + " secondary regions.")
print("The regionset has " + str(len(regionSet1.cornerRegions)) + " corner regions.")

# Example Output
# The regionset has 6 regions.
# The regionset has 0 secondary regions.
# The regionset has 0 corner regions.
