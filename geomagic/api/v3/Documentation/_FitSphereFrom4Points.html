Examples

Create a sphere that passes through 4 given points.

from geomagic.api.v3 import *


# Create points to use to define the sphere
pt1 = PointFeature()
pt1.position = Vector3D(0.0, 1.0, 0.0)

pt2 = PointFeature()
pt2.position = Vector3D(0.0, 0.0, 1.0)

pt3 = PointFeature()
pt3.position = Vector3D(0.0, 1.0, 1.0)

pt4 = PointFeature()
pt4.position = Vector3D(1.0, 0.0, 0.0)

# Create fitter
fitSphereFrom4Points = FitSphereFrom4Points()
fitSphereFrom4Points.point1 = pt1
fitSphereFrom4Points.point2 = pt2
fitSphereFrom4Points.point3 = pt3
fitSphereFrom4Points.point4 = pt4
fitSphereFrom4Points.run()

# Result sphere
sphere = fitSphereFrom4Points.resultFeature
