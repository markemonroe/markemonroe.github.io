Examples

Run VoxelIntersector.

from geomagic.api.v3 import *

vs1 = VoxelSpace()
vs1.initialize (2, 2, 2)
vs1.set (0, 0, 0, True)
vs1.set (1, 0, 0, True)
vs1.set (0, 1, 0, True)
vs1.set (0, 0, 1, True)

vs2 = VoxelSpace()
vs2.initialize (2, 2, 2)
vs2.set (1, 0, 0, True)
vs2.set (1, 0, 1, True)
vs2.set (1, 1, 0, True)
vs2.set (1, 1, 1, True)

operations = [ ]
operations.append ([VoxelIntersector.Intersect, 1])
operations.append ([VoxelIntersector.Union, 7])
operations.append ([VoxelIntersector.Subtract, 3])
operations.append ([VoxelIntersector.Flip, 6])
operations.append ([VoxelIntersector.Reverse, 4])
for op in operations:
   duplicator = Duplicate()
   duplicator.object = vs1;
   duplicator.run()
   vs3 = duplicator.clonedobject

   intersector = VoxelIntersector()
   intersector.operation = op[0]
   intersector.voxels = vs3
   intersector.other = vs2
   intersector.run()
