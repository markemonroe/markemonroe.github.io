Examples

Example of basic mesh queries

from geomagic.api.v3 import *

# Create a simple mesh
m = createTetrahedron()

# Query coordinates and normals
coord1 = m.getCoordinate(3)
normal1 = m.getTriangleNormal(3)

# Create a triangle and query its vertices
abc = OrientedTriangle(1, 0)
a = m.getVertexIndex(abc, 0)
b = m.getVertexIndex(abc, 1)
c = m.getVertexIndex(abc, 2)
print('Index: ' + str(a),str(b),str(c))

# abc can also be regarded as an edge representing ab
print('origin vertex = ' + str(m.getOrigVertex(abc)))
print('destination vertex = ' + str(m.getDestVertex(abc)))

# Query neighbors of a triangle
abc1 = m.fnext(abc)
print('origin vertex1 = ' + str(m.getOrigVertex(abc)))
print('destination vertex1 = ' + str(m.getDestVertex(abc)))
# Query a vertex star
f = 1
s = OrientedTriangleStack()
abc = OrientedTriangle(f, 0)
a = m.getVertexIndex(abc, 0)
m.getDisk(abc, s)
for i in range(0, s.getSize(), 1):
   axy = s.get(i)
   if m.getOrigVertex(axy) != a:
      print('Error')

# Transform the mesh
trans = Transform3D()
trans.setTranslation(Vector3D(1,0,0))
m.transform(trans)
coord2 = m.getCoordinate(3);
print('coord1 = ' + str(coord1.x()))
print('coord2 = ' + str(coord2.x()))
