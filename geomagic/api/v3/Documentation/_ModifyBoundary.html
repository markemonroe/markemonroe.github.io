Examples

Perform various operations to the boundary of a sample mesh.

from geomagic.api.v3 import *

# createa mesh, m, and choose an arbitrary edge, bdEf, from the boundary.
m = createRefinedTetrahedron()
tsel = TriangleSelection(m)
tsel.setSelected(74, True)

selmod = ExpandSelection(m)
selmod.triSelection = tsel
selmod.rows = 2
selmod.run();

delTri = DeleteTriangles(m)
delTri.selection = tsel
delTri.run()
esel = EdgeSelection(m)
esel.markBoundary(True)

# project the boundary containing bdEf to a plane
numBds1 = m.numBoundaryCycles
bdmod = ModifyBoundary(m)
bdmod.operation = bdmod.ProjectToPlane
plane = Plane()
plane.initialize(Vector3D(0.0, 0.0, 0.0), Vector3D(0.0, 1.0, 0.0))
bdmod.plane = plane
bdmod.tangentProjection = 0
bdmod.boundaryTriangle = esel.first()
bdmod.run()

# relax the boundary   
bdmod.operation = bdmod.Fit
bdmod.fairness = 1.0
bdmod.controlPoints = 8
bdmod.run()
bdmod.operation = bdmod.Relax
bdmod.relaxIterations = 20
bdmod.relaxStrength = 0.5
bdmod.run()
