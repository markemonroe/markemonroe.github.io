Examples

Engrave some texts on a sample mesh.

from geomagic.api.v3 import *

# Create a sample mesh m and compute some bounding box properties
m = createRefinedTetrahedron()
vmax = m.maxBoxCoord
vmaxx = vmax.x()
vc = m.centroid
vmin = m.minBoxCoord

fpv = Vector3D(vmax.x(), 0.25*vc.y()+0.75*vmin.y(), vc.z())
fp = MeshPoint(fpv)
dir = Vector3D(0, 1, 0)

# Engrave some text
engrave = Engrave(m)
engrave.offset = 0.1*(vmax.x()-vmin.x())
engrave.letterSize = 0.1*(vmax.y()-vmin.y())
engrave.origin = fp
engrave.direction = dir
engrave.font = u"Courier New Regular"
engrave.text = u"Hello, World!"
engrave.run()
