Examples

Creates a rotational surface.

from geomagic.api.v3 import *

import math

# RotationalSurface Creation
sur = RotationalSurface()

# Set Axis and the original point
sur.originPt = Vector3D(0.0, 0.0, 0.0)
sur.axisDirection = Vector3D(0.0, 0.0, 1.0)

# Create sketch
line = Line()
line.start = Vector3D (1.0, 0.0, 0.0)
line.end = Vector3D (1.0, 0.0, 3.0)

# Set Sketch
sur.sketch = line

# Set rotational angle
sur.angleFrom = 0.0
sur.angleTo = 2.0 * math.pi
