Examples

Create a sample mesh and compute its wall thickness. Report some statistics and sample value at vertex.

from geomagic.api.v3 import *

# create square with central point
mbuilder = CreateMesh()
mbuilder.mergeVertices = True
a = mbuilder.addPoint(Vector3D(0, 0, 0));
b = mbuilder.addPoint(Vector3D(-1, -1, 0));
c = mbuilder.addPoint(Vector3D(1, -1, 0));
d = mbuilder.addPoint(Vector3D(1, 1, 0));
e = mbuilder.addPoint(Vector3D(-1, 1, 0));
mbuilder.addTriangle(a, b, c)
mbuilder.addTriangle(a, c, d)
mbuilder.addTriangle(a, d, e)
mbuilder.addTriangle(a, e, b)
mbuilder.run()
mesh = mbuilder.mesh
# extrude it using shelling modifier
sheller = Shell(mesh)
height = 0.1
sheller.value = height
sheller.smooth = 0
sheller.bidirectional = 0
sheller.run()
# compute wall thickness and test it at vertex a
wtEval = ComputeWallThickness(mesh)
wtEval.maxDeviation = 1.0
wtEval.run()
maxVal = wtEval.maxThickness
minVal = wtEval.minThickness
avVal = wtEval.averageThickness
stdDevVal = wtEval.stdDevThickness
attr = wtEval.wallThickness
testVal = attr.get(a)
