Examples

Compute various geometry properties of a sample mesh.

from geomagic.api.v3 import *

# Create a sample mesh m
m = createTetrahedron()

# Create an analyzer for m
analyzer = Analyze(m)
analyzer.run()

# Geometry properties can be obtained by querying either m or analyzer
numTri = m.numTriangles
numPts = m.numPoints
isClosed = m.isClosed
numBdCycles = m.numBoundaryCycles

# Query the analyzer.
cgrav = analyzer.centerOfGravity        
area = analyzer.surfaceArea
mintriarea = analyzer.minTriangleArea
avtriarea = analyzer.averageTriangleArea
maxtriarea = analyzer.maxTriangleArea
minelen = analyzer.minEdgeLength
avelen = analyzer.averageEdgeLength
maxelen = analyzer.maxEdgeLength   
minangle = analyzer.minMinTriAngle
avangle = analyzer.averageMinTriAngle
maxangle = analyzer.maxMinTriAngle
