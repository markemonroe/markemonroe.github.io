Examples

Shows the basics of creating an UndoBlock, undoing and then redoing it.

from geomagic.api.v3 import *

pts = Points()

doc = Document()

undoManager = UndoManager()

undoManager.startUndoBlock(doc)
undoManager.addUndoObject(doc, pts, False)

# Add point
pts.addPoint(Vector3D(1.0, 2.0, 3.0))

undoManager.endUndoBlock(doc)

# pts now has 1 points so pts.numPoints should be 1 here.

# Do undo
undoManager.undo()

# Initial pts had no points so pts.numPoints should be 0 here.

# Do redo
undoManager.redo()

# pts now has 1 points so pts.numPoints should be 1 here. This is back to before the undo.
