Examples

from geomagic.api.v3 import *

# get the location of the example data
path = os.path.abspath(os.path.dirname(__file__)) + "/../../../../../pyscripts/data/"

mb = CreateMesh()
v000 = mb.addPoint(Vector3D(0.0, 0.0, 0.0))
v100 = mb.addPoint(Vector3D(1.0, 0.0, 0.0))
v010 = mb.addPoint(Vector3D(0.0, 1.0, 0.0))
v001 = mb.addPoint(Vector3D(0.0, 0.0, 1.0))
f1 = mb.addTriangle(v000, v001, v010)
f2 = mb.addTriangle(v000, v001, v100)
mb.run()
mesh = mb.mesh

saveOptions = FileSaveOptions()
saveOptions.viewDirection = Vector3D(0.0, 1.0, 0.0)
saveOptions.viewUp = Vector3D(0.0, 0.0, 1.0)
saveOptions.units = Length.Millimeters

# create write object
writer = WriteFile()
writer.filename = path + "/quadtest.ply"
writer.options = saveOptions
writer.mesh = mesh
writer.run()

# Read file back in
reader = ReadFile()
reader.filename = path + "/quadtest.ply"
reader.run()

newMesh = reader.mesh
oldCoord = mesh.getCoordinate(0)
newCoord = newMesh.getCoordinate(0)
print("old coordinate="+str(oldCoord[0])+", "+str(oldCoord[1])+", "+str(oldCoord[2]))
print("new coordinate="+str(newCoord[0])+", "+str(newCoord[1])+", "+str(newCoord[2]))
