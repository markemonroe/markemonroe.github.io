BestFitTargetLine¶

Creates a TargetLine by sliding along a given plane until it intersects with a model selection.

Examples

Fits a target line to a model.

from geomagic.api.v3 import *


# Create selection composed of a planar set of points.
pts = createPlanarPoints()
ptSelection = PointSelection(pts)
ptSelection.selectAll()

# Fit a target line to a plane
bestFitTargetLine = BestFitTargetLine()
bestFitTargetLine.selection = ptSelection
bestFitTargetLine.model = ptSelection.points
bestFitTargetLine.size = 0.05

targetPlane = Plane()
targetPlane.initialize(Vector3D(0.0, 0.0, 0.0), Vector3D(0.0, 1.0, 0.0))

# Contact the points with a cylinder that slides along the plane.
bestFitTargetLine.contactArea = BestFitTargetLine.ContactArea_Cylinder_One_Contact
bestFitTargetLine.plane = targetPlane

bestFitTargetLine.normalVector = Vector3D(0.0, 0.0, -1.0)
bestFitTargetLine.run()

# Result TargetLine.
result = bestFitTargetLine.resultFeature
