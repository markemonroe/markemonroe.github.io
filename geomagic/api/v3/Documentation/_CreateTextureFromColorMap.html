CreateTextureFromColorMap¶

Utility to create a Texture from a ColorMap.

This tool generates texture maps that can be used in Mesh or VertexObject shapes.

Examples

Create points and color them using a color map.

from geomagic.api.v3 import *

# create vertices and values
vertices = Vector3DArray()
values = DoubleArray()
NumPoints = 1000
for i in range(NumPoints):
   angle = 2.0 * math.pi * float(i) / float(NumPoints)
   vertex = Vector3D (math.sin(angle), math.cos(angle), 0.0)
   vertices.append (vertex)
   value = vertex.x() + vertex.y()
   values.append (value)

# create color map using tolerances
mod = CreateColorMapFromTolerance()
mod.toleranceMax = 1.0
mod.zeroMax = 0.1
mod.zeroMin = -0.1
mod.toleranceMin = -1.0
mod.numSegments = 11
mod.run()
colorMap = mod.resultColorMap

# create texture from color map
mod = CreateTextureFromColorMap()
mod.colorMap = colorMap
mod.run()
texture = mod.resultTexture

# create texture coordinates from color map and values
mod = CreateTextureCoordinatesFromColorMap()
mod.colorMap = colorMap
mod.values = values
mod.run()
textureCoordinates = mod.resultTextureCoordinates
