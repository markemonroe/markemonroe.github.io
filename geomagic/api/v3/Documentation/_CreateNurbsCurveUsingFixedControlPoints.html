Examples

Create NURBS curve from the interpolation points using fixed control points.

from geomagic.api.v3 import *

arr = Vector3DArray()
arr.append(Vector3D(0.0, 0.0, 0.0))
arr.append(Vector3D(1.0, 1.0, 0.0))
arr.append(Vector3D(2.0, 0.0, 0.0))
arr.append(Vector3D(3.0, 1.0, 0.0))

createNurbsCurve = CreateNurbsCurveUsingFixedControlPoints()
createNurbsCurve.pts = arr
createNurbsCurve.numControlPts = 4
createNurbsCurve.run()

curve = createNurbsCurve.nurbsCurve
