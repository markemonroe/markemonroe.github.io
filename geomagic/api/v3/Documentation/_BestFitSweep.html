bool

Input (optional)

If true, this forces the orientation of the sketch plane to occur along a World Coordinate Axis if the prior extrustion direction was within snapAngle radians of the axis. This will also force the sketch lines onto the other axes.

Examples

Fit a swept surface

from geomagic.api.v3 import *

# get the location of the example data
path = self.getParentPath(__file__, 5) + "/pyscripts/data"
# create FileRead object
filereader = ReadFile()
# set path          
filereader.filename = path + "/Easy_Arc_Line.obj"
# read the file
result = filereader.run()    
# get mesh object
mesh = filereader.mesh

# get mesh object
mesh = filereader.mesh
trmat = Transform3D()
trmat.setScaleTransform(0.001,0.001,0.001)
mesh.transform(trmat)
# Create an empty selection over m
tsel = TriangleSelection()
tsel.mesh = mesh
# Select all triangles and verify
tsel.selectAll()

self.assertNotEqual(mesh, None, "failed to read mesh model")

#Create a fitter
fitter = BestFitSweep()
fitter.meshSelection = tsel;
fitter.adaptiveSpacing = 1
fitter.fitAsLineArc = 1
fitter.localLineArcFit = 1
fitter.mergeSketches = 0
fitter.minControlPoints = 10
fitter.snap = 0
fitter.snapAngle = 0.087266
fitter.splineFitTolerance = 0.0
fitter.newApproach = 1
fitter.removeGaps = 1
fitter.smoothness = 3
fitter.run()

#get the surface
sweepSurface = fitter.resultSurface;
