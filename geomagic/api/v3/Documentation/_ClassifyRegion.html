Examples

from geomagic.api.v3 import *


# Start with a mesh of a cube with rounded edges
mesh = createRoundedBox(3)

# Break it up into regions
detect = DetectRegions()
detect.onlyPrimaryRegions = True
detect.mesh = mesh
detect.run()

# Classify each primary.
# Classification will find 6 planes. axisDirection will be the plane normal
for region in detect.regionSet.regions:
   classify = ClassifyRegion()
   classify.region = region
   classify.run()
   surfaceType = classify.surfaceType
   axisDirection = classify.axisDirection
