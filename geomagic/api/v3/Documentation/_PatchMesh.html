Examples

Warps one cube to another.

from geomagic.api.v3 import *

# create mesh
meshMain = createRoundedBox(2)
meshMain.transform (Transform3D.scale (100.0))

# delete part of the mesh
bounds = meshMain.boundingBox
selection = TriangleSelection (meshMain)
selection.selectAll ()
min = bounds.getMin()
max = bounds.getMax()
x0 = (max * 0.7 + min * 0.3).x()
x1 = (max * 1.0 + min * 0.0).x()
for tri in selection:
   a = meshMain.getVertexIndex (tri, 0, 0)
   va = meshMain.getCoordinate (a)
   b = meshMain.getVertexIndex (tri, 0, 2)
   vb = meshMain.getCoordinate (a)
   c = meshMain.getVertexIndex (tri, 0, 4)
   vc = meshMain.getCoordinate (c)
   v = (va + vb + vc) / 3
   x = v.x()
   if (x >= x0) and (x <= x1):
      selection.unselect (tri)
mod = DeleteTriangles (meshMain)
mod.selection = selection
mod.run()

# create mesh to patch
meshPatch = createRoundedBox(2)
meshPatch.transform (Transform3D.scale (100.0))

# move one mesh slightly
boundsMain = meshMain.boundingBox
diagonal = boundsMain.getMax() - boundsMain.getMin()
shift = diagonal * 0.02
meshPatch.transform (Transform3D.translation (shift))
overlapDistance = 2.0 * shift.length()

# perform the warp
mod = WarpToMesh()
mod.mesh = meshPatch
mod.meshTarget = meshMain
mod.overlapDistance = overlapDistance
mod.run()

# patch mesh
mod = PatchMesh()
mod.mesh = meshMain
mod.meshPatch = meshPatch
mod.overlapDistance = overlapDistance
mod.relaxIterations = 1
mod.run()
