Examples

from geomagic.api.v3 import *

axis = Vector3D(0.5, 0.5, 0.8)
axis.normalize()
rotmat = Transform3D()

# create rotation transform around axis
rotmat.setRotationTransform(axis, 0.2) 
trmat = Transform3D()
translation = Vector3D(0.0, 0.0, 1.0)

# create translation transform
trmat.setTranslationTransform(translation) 
xf = Transform3D(rotmat, trmat) 

# create full transformation
coord = Vector3D(0.1, 0.5, 0.2)

# apply full transformation to coord
trcoord = xf.mult(coord)
