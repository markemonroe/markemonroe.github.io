ModifyPolylineSet¶

Perform polyline set operations: project, resample, fit, relax, verify and Convert. This class is experimental for the current release and will be enhanced in future releases.

Examples

Modify polyline set

from geomagic.api.v3 import *

# Create a polyleSet
(m,polySet,polyId1) = simplePolylineSet()

# Modify mwf by projecting
mod = ModifyPolylineSet(m)
mod.polylineSet = polySet
mod.operation = mod.Project
result = mod.run()

# Modify mwf by resampling
mod.currentPath = polyId1
mod.operation = mod.Resample
mod.resampleSize = 10
mod.resampleDistance = 0.0 #use resample size only
mod.run()

# Modify mwf by fitting
mod.operation = mod.Fit
mod.fittingControlPoints = 6
mod.fittingFairness = 0.5
mod.run()

# Modify mwf by relaxation
mod.operation = mod.Relax
mod.relaxIterations = 100
mod.run()
