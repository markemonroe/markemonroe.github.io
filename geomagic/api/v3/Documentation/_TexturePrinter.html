Examples

Print text into a texture.

from geomagic.api.v3 import *

printer = TexturePrinter()
printer.font = "Courier New"
printer.points = 48
printer.text = "Hello World!"
printer.color = GeoColor (0.0, 0.0, 0.0)
printer.background = GeoColor (1.0, 1.0, 1.0, 0.0)
printer.run()
