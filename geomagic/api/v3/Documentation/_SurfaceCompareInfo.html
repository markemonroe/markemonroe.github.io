SurfaceCompareInfo¶

The result data from a CompareMeshToSurface. Contains per-vertex distances and statistics.

Examples

from geomagic.api.v3 import *


# Create a cylinder to convert.
cylinder = Cylinder()
cylinder.initialize(Vector3D(0.0, 0.0, 0.0), Vector3D(0.0, 0.0, 1.0), 2.0, 2.5)

# Create amd run modifier to convert cylinder to a mesh.
createMeshFromSurface = CreateMeshFromSurface()
createMeshFromSurface.surface = cylinder
createMeshFromSurface.run()

# Mesh that approximates the surface of the cylinder.
mesh = createMeshFromSurface.mesh

# compare the mesh aproximation to the cylinder
compareMod = CompareMeshToSurface()
sel = TriangleSelection()
sel.mesh = mesh
sel.selectAll()
compareMod.selection = sel
compareMod.surface = cylinder
compareMod.run()

compareStats = compareMod.comparisonResult

# For the comparison, only the sides of the cylinder are used and not the caps.
# Therefore, vertices that are too far from the sides of the cylinder will be part of
# the uncomparedVertices vector and not be included in the distance array or statistics.
comparedVertices = compareStats.comparedVertices
uncomparedVertices = compareStats.uncomparedVertices
distanceValues = compareStats.distanceValues
