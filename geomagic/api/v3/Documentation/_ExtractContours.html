Examples

Extract contours from selection with required foldVector and separatorSelection input. Always Work with ComputeCurvatureAttribute and ComputeSeparator modifiers.

from geomagic.api.v3 import *

# Create a mesh model m and compute its curvature attributes.
m = createTetrahedron()
curveAttr = ComputeCurvatureAttribute(m)
curveAttr.smoothing = 2
curveAttr.run()

# Query the curvature attributes of mesh.
attr = curveAttr.curvatureAttribute
foldVec = curveAttr.foldVector
subdivsz = curveAttr.defaultSeparatorSubdivisionSize
spacing = curveAttr.defaultExtractSpacing
minlen = curveAttr.defaultExtractMinimumLength

# Compute a separator selection tsel using curvature attributes above.
# Set curvatureSensitivity as 0.3.
computeSep = ComputeSeparator(m)
computeSep.foldVector = foldVec
computeSep.curvatureAttribute = attr
computeSep.curvatureSensitivity = 0.3
computeSep.separatorSubdivisionSize = subdivsz
computeSep.run()
tsel = computeSep.separatorSelection

# Query the number of selected triangles. 
numSelTri = tsel.numSelected

# Extract a set of curves from the selection tsel not including boundary.
extractor = ExtractContours(m)
extractor.foldVector = foldVec
extractor.curvatureAttribute = attr
extractor.separatorSelection = tsel
extractor.includeBoundary = 0
extractor.spacing = spacing
extractor.minimumLength = minlen
extractor.run()

# Store the result as a MeshWireFrame and query the number of paths and points.
polySet = extractor.resultContours 
numPolylines = polySet.numPolylines
