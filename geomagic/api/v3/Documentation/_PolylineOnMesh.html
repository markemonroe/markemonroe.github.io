Examples

Create a simple Polyline

from geomagic.api.v3 import *

polyline = PolylineOnMesh()
polyline.initialize(3)

# set the points
polyline.points[0] = Vector3D(0.0, 0.0, 0.0)
polyline.points[1] = Vector3D(1.0, 1.0, 0.0)
polyline.points[2] = Vector3D(2.0, 0.0, 0.0)

# set the knots
polyline.knots[0] = 0.0
polyline.knots[1] = 0.5
polyline.knots[2] = 1.0

# set the triIndexes
polyline.triIndexes[0] = 1
polyline.triIndexes[1] = 2
polyline.triIndexes[2] = 3

polyline.range = Vector2D(0.0, 1.0)

# Evaluate point along the polyline.
v = polyline.evaluatePoint(0.2)

# Make transform that rotates the polyline to lie along y axis
transform = Transform3D()
transform.setRotationTransform(Vector3D(1.0, 0.0, 0.0), Vector3D(0.0, 1.0, 0.0))

# Transform
polyline.transform(transform)
