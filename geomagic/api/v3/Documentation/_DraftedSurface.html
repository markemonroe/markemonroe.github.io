Examples

Create a drafted surface

from geomagic.api.v3 import *

import math

# DraftedSurface Creation
sur = DraftedSurface()

# Create sketch
arc = Arc()
arc.center = Vector3D (0.0, 0.0, 0.5)
arc.radius = 0.5
arc.xAxis = Vector3D (0.0, 0.0, 1.0)
arc.normal = Vector3D (0.0, 0.0, 1.0)
arc.startAngle = 0.0
arc.endAngle = 2.0 * math.pi

# Set Axis, draftedAngle and Height
sur.extrudeDir = Vector3D (0.0, 0.0, 1.0)
sur.sketch = arc

sur.draftAngle = 45.0/180 * math.pi
sur.heightFrom = -0.5
sur.heightTo = 0.5
