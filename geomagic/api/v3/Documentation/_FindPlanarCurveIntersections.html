Examples

from geomagic.api.v3 import *
line1 = Line()
line1.start = Vector3D (0.0, 0.0, 0.0)
line1.end = Vector3D (0.0, 1.0, 0.0)

# lineCurve creation
line2 = Line()
line2.start = Vector3D (0.5, 0.5, 0)
line2.end = Vector3D (-0.5, 0.5, 0)

# Plane Feature Creation
plane1 = Plane()
plane1.normal = Vector3D(0.0, 0.0, 1.0)
plane1.origin = Vector3D(0.0, 0.0, 0.0)

# Find the intersections
findintersection = FindPlanarCurveIntersections()
findintersection.curve1 = line1
findintersection.curve2 = line2
findintersection.plane = plane1
findintersection.run()
