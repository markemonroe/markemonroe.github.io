Examples

Section points

from geomagic.api.v3 import *

pts = Points()
pts.addPoint(Vector3D(0.0, 0.0, 0.0))
pts.addPoint(Vector3D(0.5, 0.5, 0.5))
pts.addPoint(Vector3D(1.5, 1.5, 1.5))
plane = Plane()
plane.initialize(Vector3D(0.0, 0.0, 0.5), Vector3D(0.0, 0.0, 1.0))
sectioner = SectionPoints()
sectioner.points = pts
sectioner.thickness = 0.1
sectioner.plane = plane
sectioner.run()
sectionpts = sectioner.section
numSectionPoints = sectionpts.numPoints
polySet = sectioner.polylineSet
