Examples

Evaluate types of all model faces.

from geomagic.api.v3 import *
from geomagic.api.v3.cad import *
path = Directories.getDirectory(Directories.DIRECTORY_EXAMPLE_DATA)

# Load CAD model from file
reader = ReadFile()
reader.filename = path + "/cube.igs"
reader.run()
model = reader.cadModel

# Create feature evaluator object
evaluator = CreateSurfaceFromCADFace()
evaluator.cadModel = model

# Loop through all model faces and print their feature types
selection = CADFaceSelection()
selection.cadModel = model
selection.selectAll()

for faceId in selection:
   evaluator.faceId = faceId
   evaluator.run()
   
   feat = evaluator.feature
   featType = evaluator.featureType
   
   print("Type of face", faceId, "is", featType)
