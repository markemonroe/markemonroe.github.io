Examples

Create a point cover for a sample cad.

from geomagic.api.v3 import *
from geomagic.api.v3.cad import *

converter = CreateCADFromSurface()
converter.surface = Cylinder()
converter.run()
model = converter.cadModel

#create point cover for a whole model
cover = CreatePointsFromCAD()
cover.cadModel = model
cover.offset = 0.001
cover.targetNumPoints = 1000
cover.run()

points = cover.points
